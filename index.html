using UnityEngine; using UnityEngine.UI; // ===================== GAME CONTROLLER ====================== public class GameController : MonoBehaviour { [Header("UI")] public Text questionText; public Button[] answerButtons; public Text levelText; [Header("Player")] public AudioSource shootSound; public GameObject bulletPrefab; public Transform gunPoint; [Header("Monsters")] public GameObject monsterNormalPrefab; public GameObject monsterBossPrefab; public Transform monsterSpawnPoint; public Transform playerPos; private GameObject currentMonster; private int monsterHealth; private int level = 1; private int correctAnswer; public static float monsterSpeed = 1.2f; // سرعة الوحش المختارة من اللاعب void Start() { StartLevel(); } void StartLevel() { levelText.text = "المرحلة: " + ArabicNumber(level); SpawnMonster(); GenerateQuestion(); } void SpawnMonster() { if (level % 3 == 0) { currentMonster = Instantiate(monsterBossPrefab, monsterSpawnPoint.position, Quaternion.identity); monsterHealth = 3; } else { currentMonster = Instantiate(monsterNormalPrefab, monsterSpawnPoint.position, Quaternion.identity); monsterHealth = 1; } currentMonster.AddComponent().Init(playerPos, monsterSpeed, this); } public void MonsterReachedPlayer() { Time.timeScale = 0; questionText.text = "انتهت اللعبة!"; } public void HitMonster() { monsterHealth--; if (monsterHealth <= 0) { Destroy(currentMonster); level++; StartLevel(); } } void GenerateQuestion() { int x = Random.Range(1, 11); correctAnswer = 4 * x; questionText.text = "٤ × " + ArabicNumber(x); int correctIndex = Random.Range(0, 3); answerButtons[correctIndex].GetComponentInChildren().text = ArabicNumber(correctAnswer); for (int i = 0; i < 3; i++) { if (i == correctIndex) continue; int wrong = correctAnswer + Random.Range(1, 5) * (Random.value > 0.5f ? 1 : -1); if (wrong <= 0) wrong = 1; answerButtons[i].GetComponentInChildren().text = ArabicNumber(wrong); } foreach (Button btn in answerButtons) { btn.onClick.RemoveAllListeners(); btn.onClick.AddListener(() => CheckAnswer(btn)); } } void CheckAnswer(Button btn) { int chosen = ParseArabic(btn.GetComponentInChildren().text); if (chosen == correctAnswer) { shootSound.Play(); Instantiate(bulletPrefab, gunPoint.position, Quaternion.identity); HitMonster(); } GenerateQuestion(); } string ArabicNumber(int n) { return n.ToString() .Replace('0', '٠') .Replace('1', '١') .Replace('2', '٢') .Replace('3', '٣') .Replace('4', '٤') .Replace('5', '٥') .Replace('6', '٦') .Replace('7', '٧') .Replace('8', '٨') .Replace('9', '٩'); } int ParseArabic(string s) { return int.Parse( s.Replace("٠", "0") .Replace("١", "1") .Replace("٢", "2") .Replace("٣", "3") .Replace("٤", "4") .Replace("٥", "5") .Replace("٦", "6") .Replace("٧", "7") .Replace("٨", "8") .Replace("٩", "9") ); } } // ===================== حركة الوحش ====================== public class MonsterMover : MonoBehaviour { private Transform player; private float speed; private GameController controller; public void Init(Transform p, float s, GameController gc) { player = p; speed = s; controller = gc; } void Update() { transform.position = Vector3.MoveTowards(transform.position, player.position, speed * Time.deltaTime); if (Vector3.Distance(transform.position, player.position) < 0.1f) { controller.MonsterReachedPlayer(); } } }
